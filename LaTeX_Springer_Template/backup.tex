\begin{split}
r'_{i,j} = r_{i,j} + s \cdot l \cdot w_{i,j} \\
d(r_{i,j}) = \begin{cases}
\rho' - \rho_{0} &  d < r_{i} \\
0 &  d > r_{o} \\
